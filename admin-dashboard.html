📄 VerifyContract ABI (Solidity)

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract VerifyContract {
  event DegreeVerified(string degreeId, address employer, bool valid);
  event FraudDetected(string degreeId, address employer);

  function verifyDegree(
    string memory degreeId,
    bytes32 submittedHash
  ) public returns (bool) {
    Degree memory d = DegreeContract.getDegree(degreeId);
    if (d.degreeHash != submittedHash) {
      emit FraudDetected(degreeId, msg.sender);
      return false;
    }
    emit DegreeVerified(degreeId, msg.sender, true);
    return true;
  }
}
